- Wind Turbine Management:
	Schema DetailTurbine:
		id: int
		name: string
		latitude: float
		longitude: float

	Schema CreateTurbine:
		name: string
		latitude: float
		longitude: float
	
	get: /turbines/{id}
		response 200:
			DetailTurbine
		response 404: not found
	
	get: /turbines
		response 200:
			DetailTurbine[]
	
	post: /turbines
		body:
			CreateTurbine
		response 200:
			DetailTurbine
	
	patch: /turbines/{id}
		body: 
			CreateTurbine
		response 200:
			DetailTurbine
		response 404: not found
	
	delete: /turbines/{id}
		response 200: no Content 

- Task Management:
	Schema DetailTask:
		id: int
		name: string
		durationHours: double/ float
		finished: bool
		earliestStart: string (ISO-8601)
		latestFinish: string (ISO-8601)
		requiredQualifications: TaskQualification[] (siehe aktuelles Backend)
		requiredTools: TaskTool[] (siehe aktuelles Backend)
		requiredTurbine: int (id of the turbine)

	
	Schema SummaryTask: 
		id: int,
		name: string,
		durationHours: double/ float
		finished: bool
		
	Schema CreateTask:
		name: string
		durationHours: double/ float
		finished: bool
		earliestStart: string (ISO-8601)
		latestFinish: string (ISO-8601)
		requiredQualifications: TaskQualification[]
		requiredTools: TaskTool[]
		location: int
		
	get: /tasks/{id}
		response 200:
			DetailTask
		response 404: not found
	
	get: /task?done=true&after={date}&before={date}
		response 200:
			SummaryTask[]
	
	post: /tasks
		body:
			CreateTask
		response 200:
			DetailTask
	
	patch: /tasks/{id}
		body: 
			CreateTask
		response 200:
			DetailTurbine
		response 404: not found
	
	delete: /tasks/{id}
		response 200: no Content 
		
- GMPL Plan:
	Schema BoatTool:
		tool : TaskTool
		requiredAmount : int

	Schema BoatSchedule:
		departure : TimeOnly
		arrival : TimeOnly

	Schema TaskSchedule:
		task : TaskSummary
		startTime : TimeOnly

	Schema BoatPlan:
		persons: SummaryPerson[]
		tools : BoatTool[]
		taskSchedules : TaskSchedule[]
		boatSchedules : BoatSchedules[]

	Schema Plan:
		date : DateOnly (ISO-8601)
		createdAt : DateTime
		boats : BoatPlan[]

	post: /plan
		body:
			maxTime: float/ double
			boatNumber: int
			date: string (ISO-8601)
			speed: double/ float
		response 200:
			totalTime: double/float
			boats[]: 
				taskItems[]: 
					SummaryTask
					travelTime: double/float
					startTime: string (ISO-8601)
				persons: SummaryPerson[]
				tools: TaskTool[]
				
	get: /plan/{date}
		response 200:
			totalTime: double/float
			boats[]: 
				taskItems[]: 
					SummaryTask
					travelTime: double/float
					startTime: string (ISO-8601)
				persons: SummaryPerson[]
				tools: TaskTool[]
		response 404: not found
		
	delete: /plan/{date}
		response 200: no Content
